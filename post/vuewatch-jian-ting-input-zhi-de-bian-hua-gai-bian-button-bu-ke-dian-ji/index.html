[vue]watch 监听input值的变化 改变button不可点击

2019-05-27
	<div id="app">
        <input v-model="test">
        <button :disabled="isdisabled">测试</button>
    </div>

	var app = new Vue({
        el: "#app",
        data: {
            test:123,
            isdisabled: false,
        },
        watch:{
				//test(){} 和test:function(){} 效果一样
            test:function(){
                if(this.test==1234){
                    this.isdisabled=true;
                }
            }
        }
    });
